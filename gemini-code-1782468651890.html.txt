    <script>
        // Mengambil nama tamu dari URL (?to=Nama+Tamu)
        const urlParams = new URLSearchParams(window.location.search);
        const namaTamu = urlParams.get('to') || "Tamu Undangan";
        document.getElementById('tamu-undangan').innerText = namaTamu;

        const music = document.getElementById('bg-music');

        function bukaUndangan() {
            document.getElementById('cover').style.display = 'none';
            document.getElementById('main-content').style.display = 'block';
            music.play().catch(error => console.log("Autoplay dicegah browser, musik aktif lewat interaksi."));
        }

        function toggleMusic() {
            if (music.paused) {
                music.play();
                document.getElementById('music-control').innerText = "🎵";
            } else {
                music.pause();
                document.getElementById('music-control').innerText = "🔇";
            }
        }

        function salinNorek() {
            const norek = document.getElementById('norek').innerText;
            navigator.clipboard.writeText(norek);
            alert("Nomor rekening berhasil disalin!");
        }

        function kirimWA(event) {
            event.preventDefault();
            const nama = document.getElementById('nama-rsvp').value;
            const status = document.getElementById('status-hadir').value;
            const noHP = "628123456789"; // Ganti dengan nomor WhatsApp Anda
            const teks = `Halo, saya ${nama} menyatakan *${status}* dalam acara Tasyakuran Khitanan Pandu.`;
            window.open(`https://api.whatsapp.com/send?phone=${noHP}&text=${encodeURIComponent(teks)}`, '_blank');
        }
    </script>
